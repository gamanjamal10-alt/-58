rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Products: Public can read, only admin can write
    match /products/{productId} {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Delivery Fees: Public can read, only admin can write
    // All fees are stored in a single document for efficiency
    match /config/deliveryFees {
      allow read: if true;
      allow write: if request.auth != null;
    }

    // Orders: Public can create, only admin can read/update
    match /orders/{orderId} {
      allow create: if true; // Anyone can place an order
      allow read, update, delete: if request.auth != null; // Only admin can manage orders
    }
  }
}
